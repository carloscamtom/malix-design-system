## Installation

```tsx
import { Breadcrumb } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { Breadcrumb } from '@camtomlabs/malix-design-system';
import { Home } from 'lucide-react';

export function PageBreadcrumb() {
  return (
    <Breadcrumb
      items={[
        { label: 'Home', href: '/', icon: <Home size={14} /> },
        { label: 'Projects', href: '/projects' },
        { label: 'Current Project' },
      ]}
    />
  );
}
```
