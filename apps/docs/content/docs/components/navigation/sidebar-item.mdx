## Installation

```tsx
import { SidebarItem } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { SidebarItem } from '@camtomlabs/malix-design-system';
import { Home, Settings } from 'lucide-react';

export function Navigation() {
  return (
    <nav>
      <SidebarItem icon={<Home size={18} />} active>
        Home
      </SidebarItem>
      <SidebarItem icon={<Settings size={18} />}>
        Settings
      </SidebarItem>
    </nav>
  );
}
```
