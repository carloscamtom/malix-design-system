## Installation

```tsx
import { SidebarPanel } from '@malix/ui';
```

## Usage

```tsx
import { useState } from 'react';
import { SidebarPanel, SidebarItem } from '@malix/ui';
import { Home, Settings, HelpCircle } from 'lucide-react';

export function AppSidebar() {
  const [collapsed, setCollapsed] = useState(false);

  return (
    <SidebarPanel
      collapsed={collapsed}
      onToggleCollapse={() => setCollapsed((c) => !c)}
      logo={<span className="text-white font-bold">Malix</span>}
      collapsedLogo={<span className="text-white font-bold">M</span>}
      navigation={
        <>
          <SidebarItem icon={<Home size={18} />} active>
            Home
          </SidebarItem>
          <SidebarItem icon={<Settings size={18} />}>
            Settings
          </SidebarItem>
        </>
      }
      footer={
        <SidebarItem icon={<HelpCircle size={18} />}>
          Help & Support
        </SidebarItem>
      }
    />
  );
}
```
