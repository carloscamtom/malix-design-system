## Installation

```tsx
import { SplitPane } from '@malix/ui';
```

## Usage

```tsx
import { SplitPane } from '@malix/ui';

export function EditorLayout() {
  return (
    <SplitPane
      leftTitle="Editor"
      rightTitle="Preview"
      leftPanel={
        <div style={{ padding: 16 }}>
          <p>Write your content here.</p>
        </div>
      }
      rightPanel={
        <div style={{ padding: 16 }}>
          <p>Live preview of the output.</p>
        </div>
      }
      defaultSplit={50}
    />
  );
}
```
