## Installation

```tsx
import { Accordion } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
<Accordion title="What payment methods do you accept?">
  <p>We accept all major credit cards, PayPal, and bank transfers.</p>
</Accordion>
```

## Default Open

Render the accordion expanded on mount:

```tsx
<Accordion title="Getting Started" defaultOpen>
  <p>Follow the installation guide to set up your project.</p>
</Accordion>
```

## With Icon

Add a leading icon to the accordion header:

```tsx
import { HelpCircle } from 'lucide-react';

<Accordion title="Frequently Asked Questions" icon={<HelpCircle size={16} />}>
  <p>Find answers to common questions below.</p>
</Accordion>
```
