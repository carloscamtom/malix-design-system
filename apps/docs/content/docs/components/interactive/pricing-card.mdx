## Installation

```tsx
import { PricingCard } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
<PricingCard
  planName="Pro"
  price="$29"
  description="For growing teams that need more power."
  features={[
    'Unlimited projects',
    '50 GB storage',
    'Priority support',
    'Advanced analytics',
  ]}
  onCtaClick={() => handleSubscribe('pro')}
/>
```

## Highlighted Plan

Draw attention to a recommended plan:

```tsx
<PricingCard
  planName="Business"
  price="$79"
  period="/month"
  description="Best for scaling organizations."
  features={[
    'Everything in Pro',
    'Unlimited storage',
    'SSO & SAML',
    'Dedicated account manager',
    'Custom integrations',
  ]}
  ctaLabel="Start Free Trial"
  onCtaClick={() => handleSubscribe('business')}
  highlighted
/>
```

## Custom Period

```tsx
<PricingCard
  planName="Starter"
  price="$99"
  period="/year"
  features={['5 projects', '10 GB storage', 'Email support']}
  ctaLabel="Subscribe"
/>
```
