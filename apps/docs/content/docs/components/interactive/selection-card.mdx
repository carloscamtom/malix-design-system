## Installation

```tsx
import { SelectionCard } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { Zap } from 'lucide-react';

const [selected, setSelected] = useState('fast');

<SelectionCard
  title="Fast Delivery"
  description="Get your order within 24 hours."
  icon={<Zap size={20} />}
  active={selected === 'fast'}
  onClick={() => setSelected('fast')}
/>
```

## Card Group

Use multiple SelectionCards together for single-select or multi-select patterns:

```tsx
import { Globe, Shield, Zap } from 'lucide-react';

<div style={{ display: 'flex', gap: 12 }}>
  <SelectionCard
    title="Standard"
    description="5-7 business days"
    icon={<Globe size={20} />}
    active={selected === 'standard'}
    onClick={() => setSelected('standard')}
  />
  <SelectionCard
    title="Express"
    description="2-3 business days"
    icon={<Zap size={20} />}
    active={selected === 'express'}
    onClick={() => setSelected('express')}
  />
  <SelectionCard
    title="Secure"
    description="Tracked and insured"
    icon={<Shield size={20} />}
    active={selected === 'secure'}
    onClick={() => setSelected('secure')}
  />
</div>
```
