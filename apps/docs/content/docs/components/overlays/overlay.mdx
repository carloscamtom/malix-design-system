## Installation

```tsx
import { Overlay } from '@malix/ui';
```

## Usage

```tsx
const [open, setOpen] = useState(false);

<Overlay
  open={open}
  title="Settings"
  onClose={() => setOpen(false)}
>
  <p>Configure your preferences below.</p>
</Overlay>
```
