## Installation

```tsx
import { Dropzone } from '@malix/ui';
```

## Usage

```tsx
<Dropzone
  onDrop={(files) => handleUpload(files)}
  accept="image/*"
/>
```

## Custom Labels

Override the default title and hint text:

```tsx
<Dropzone
  onDrop={handleUpload}
  title="Upload your logo"
  hint="PNG or SVG, max 2 MB"
  accept=".png,.svg"
/>
```

## Disabled State

Prevent interaction while an upload is in progress:

```tsx
<Dropzone
  onDrop={handleUpload}
  disabled
  title="Upload in progress..."
  hint="Please wait"
/>
```

## Accept Specific Types

Restrict the accepted file types:

```tsx
<Dropzone
  onDrop={handleUpload}
  accept=".pdf,.docx,.xlsx"
  title="Drop documents here"
  hint="PDF, Word, or Excel files"
/>
```
