## Installation

```tsx
import { AIAssistantPanel } from '@camtomlabs/malix-design-system';
import type { AIAssistantMessage } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
const [messages, setMessages] = useState<AIAssistantMessage[]>([
  { id: '1', variant: 'ai', content: 'Hi! How can I help?' },
]);

<AIAssistantPanel
  messages={messages}
  onSend={(text) => {
    setMessages((prev) => [
      ...prev,
      { id: String(Date.now()), variant: 'user', content: text },
    ]);
  }}
  onClose={() => setOpen(false)}
  userInitials="CG"
/>
```

## Custom Title

```tsx
<AIAssistantPanel
  title="Support Bot"
  messages={messages}
  onSend={handleSend}
  placeholder="Describe your issue..."
/>
```

## Disabled State

```tsx
<AIAssistantPanel
  messages={messages}
  onSend={handleSend}
  disabled
  placeholder="AI is processing..."
/>
```
