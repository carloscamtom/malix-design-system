## Installation

```tsx
import { DataTable } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
const columns = [
  { key: 'name', header: 'Name', width: '200px' },
  { key: 'role', header: 'Role' },
  { key: 'status', header: 'Status', render: (row) => (
    <Badge variant={row.status === 'Active' ? 'success' : 'neutral'}>
      {row.status}
    </Badge>
  )},
];

const data = [
  { id: 1, name: 'Alice Johnson', role: 'Engineer', status: 'Active' },
  { id: 2, name: 'Bob Smith', role: 'Designer', status: 'Inactive' },
];

<DataTable
  columns={columns}
  data={data}
  onRowClick={(row) => navigate(`/users/${row.id}`)}
/>
```

## Empty State

When no data is available, a message is displayed:

```tsx
<DataTable
  columns={columns}
  data={[]}
  emptyMessage="No users found matching your search."
/>
```
