## Installation

```tsx
import { Pagination } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
const [page, setPage] = useState(1);

<Pagination
  currentPage={page}
  totalPages={10}
  onPageChange={setPage}
/>
```

## Variants

### Full

The default variant displays page numbers with previous/next navigation:

```tsx
<Pagination
  variant="full"
  currentPage={3}
  totalPages={10}
  onPageChange={setPage}
/>
```

### Mini

A compact variant showing only the current page indicator and navigation arrows:

```tsx
<Pagination
  variant="mini"
  currentPage={3}
  totalPages={10}
  onPageChange={setPage}
/>
```
