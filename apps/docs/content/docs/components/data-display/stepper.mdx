## Installation

```tsx
import { Stepper } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { User, CreditCard, Check } from 'lucide-react';

<Stepper
  steps={[
    { label: 'Account', icon: <User size={16} />, status: 'completed' },
    { label: 'Payment', icon: <CreditCard size={16} />, status: 'active' },
    { label: 'Confirmation', icon: <Check size={16} />, status: 'pending' },
  ]}
/>
```

## Step Statuses

Each step can have one of three statuses:

```tsx
<Stepper
  steps={[
    { label: 'Details', status: 'completed' },
    { label: 'Review', status: 'active' },
    { label: 'Publish', status: 'pending' },
  ]}
/>
```
