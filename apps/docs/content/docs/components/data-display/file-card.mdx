## Installation

```tsx
import { FileCard } from '@malix/ui';
```

## Usage

```tsx
import { FileText } from 'lucide-react';

<FileCard
  fileName="annual-report.pdf"
  meta="2.4 MB"
  icon={<FileText size={20} />}
  onAction={() => handleDownload('annual-report.pdf')}
/>
```

## Without Icon

The file card works without a custom icon:

```tsx
<FileCard
  fileName="design-tokens.json"
  meta="12 KB"
/>
```
