## Installation

```tsx
import { OperationStatus } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { OperationStatus } from '@camtomlabs/malix-design-system';

export function DeploymentRow() {
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
      <OperationStatus status="completed" />
      <span>Deployment finished</span>
    </div>
  );
}
```

## Variants

### Active

```tsx
<OperationStatus status="active" />
```

### Completed

```tsx
<OperationStatus status="completed" />
```

### Failed

```tsx

```

### Pending

```tsx
<OperationStatus status="pending" />
```
