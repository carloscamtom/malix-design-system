## Installation

```tsx
import { Tooltip } from '@malix/ui';
```

## Usage

```tsx
import { Tooltip, Button } from '@malix/ui';

export function SaveAction() {
  return (
    <Tooltip content="Save your changes" placement="top">
      <Button hierarchy="primary">Save</Button>
    </Tooltip>
  );
}
```

## Placements

### Top

```tsx
<Tooltip content="Tooltip on top" placement="top">
  <Button hierarchy="secondary">Top</Button>
</Tooltip>
```

### Bottom

```tsx
<Tooltip content="Tooltip on bottom" placement="bottom">
  <Button hierarchy="secondary">Bottom</Button>
</Tooltip>
```

### Left

```tsx
<Tooltip content="Tooltip on left" placement="left">
  <Button hierarchy="secondary">Left</Button>
</Tooltip>
```

### Right

```tsx
<Tooltip content="Tooltip on right" placement="right">
  <Button hierarchy="secondary">Right</Button>
</Tooltip>
```
