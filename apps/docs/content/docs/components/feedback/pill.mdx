## Installation

```tsx
import { Pill } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { Pill } from '@camtomlabs/malix-design-system';

export function TagList() {
  return (
    <div style={{ display: 'flex', gap: 8 }}>
      <Pill variant="primary">Feature</Pill>
      <Pill variant="success">Shipped</Pill>
      <Pill variant="warning">In Review</Pill>
    </div>
  );
}
```
