## Installation

```tsx
import { ProgressBar } from '@malix/ui';
```

## Usage

```tsx
import { ProgressBar } from '@malix/ui';

export function UploadProgress() {
  return <ProgressBar value={65} label="Uploading files" />;
}
```

## Variants

### Default

```tsx
<ProgressBar value={50} label="Processing" />
```

### Success

```tsx
<ProgressBar value={100} label="Complete" variant="success" />
```

### Without Percentage

```tsx
<ProgressBar value={30} label="Loading" showPercent={false} />
```
