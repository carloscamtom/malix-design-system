## Installation

```tsx
import { Input } from '@malix/ui';
```

## Usage

```tsx
<Input placeholder="Enter your name" />
```

## With Leading Icon

Display an icon inside the input to provide visual context:

```tsx
import { Search } from 'lucide-react';

<Input
  placeholder="Search..."
  leadingIcon={<Search size={16} />}
/>
```
