## Installation

```tsx
import { ChatInput } from '@malix/ui';
```

## Usage

```tsx
const [message, setMessage] = useState('');

<ChatInput
  value={message}
  onChange={(value) => setMessage(value)}
  onSend={() => {
    console.log('Sent:', message);
    setMessage('');
  }}
  placeholder="Type a message..."
/>
```

## Custom Placeholder

```tsx
<ChatInput
  value={message}
  onChange={(value) => setMessage(value)}
  onSend={() => handleSend()}
  placeholder="Ask a question..."
/>
```

## Disabled State

```tsx
<ChatInput
  value=""
  onChange={() => {}}
  onSend={() => {}}
  placeholder="Chat is disabled"
  disabled
/>
```
