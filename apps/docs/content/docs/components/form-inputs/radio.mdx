## Installation

```tsx
import { Radio } from '@malix/ui';
```

## Usage

```tsx
<Radio
  label="Option A"
  name="example"
  value="a"
  checked={true}
  onChange={(value) => console.log(value)}
/>
```

## Radio Group

Combine multiple Radio components with the same `name` to form a group:

```tsx
<div>
  <Radio name="plan" value="free" label="Free" onChange={(value) => console.log(value)} />
  <Radio name="plan" value="pro" label="Pro" onChange={(value) => console.log(value)} />
  <Radio name="plan" value="enterprise" label="Enterprise" onChange={(value) => console.log(value)} />
</div>
```

## Disabled State

```tsx
<Radio name="locked" value="selected" label="Selected (locked)" checked disabled />
<Radio name="locked" value="unselected" label="Unselected (locked)" disabled />
```
