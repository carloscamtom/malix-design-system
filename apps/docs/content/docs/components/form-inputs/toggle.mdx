## Installation

```tsx
import { Toggle } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
<Toggle
  label="Enable notifications"
  checked={false}
  onChange={(checked) => console.log(checked)}
/>
```

## Checked State

```tsx
<Toggle
  label="Dark mode"
  checked={true}
  onChange={(checked) => console.log(checked)}
/>
```

## Disabled State

```tsx
<Toggle label="Maintenance mode" checked={false} disabled />
<Toggle label="Always on" checked={true} disabled />
```
