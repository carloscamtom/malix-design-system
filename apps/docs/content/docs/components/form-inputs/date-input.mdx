## Installation

```tsx
import { DateInput } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
<DateInput
  placeholder="dd/mm/yyyy"
  onChange={(value) => console.log(value)}
/>
```

## With Default Value

```tsx
<DateInput
  value="2025-03-15"
  onChange={(value) => console.log(value)}
/>
```

## Disabled State

```tsx
<DateInput
  value="2025-01-01"
  disabled
/>
```
