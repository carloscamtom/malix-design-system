## Installation

```tsx
import { SelectGroup, Select } from '@malix/ui';
```

## Usage

```tsx
<SelectGroup label="Country" helperText="Select your country of residence.">
  <Select
    placeholder="Choose a country"
    options={[
      { label: 'United States', value: 'us' },
      { label: 'Canada', value: 'ca' },
      { label: 'Mexico', value: 'mx' },
    ]}
  />
</SelectGroup>
```

## Error State

Display a validation error by setting the `error` prop:

```tsx
<SelectGroup label="Department" error helperText="Please select a department.">
  <Select
    placeholder="Choose a department"
    options={[
      { label: 'Engineering', value: 'engineering' },
      { label: 'Design', value: 'design' },
      { label: 'Product', value: 'product' },
    ]}
  />
</SelectGroup>
```
