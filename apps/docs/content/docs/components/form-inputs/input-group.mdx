## Installation

```tsx
import { InputGroup } from '@malix/ui';
```

## Usage

```tsx
<InputGroup label="Email address" helperText="We will never share your email.">
  <Input placeholder="you@example.com" />
</InputGroup>
```

## With Leading Icon

```tsx
import { Mail } from 'lucide-react';

<InputGroup
  label="Email"
  helperText="Enter your work email"
  leadingIcon={<Mail size={16} />}
>
  <Input placeholder="you@company.com" />
</InputGroup>
```

## Error State

Display a validation error by setting the `error` prop:

```tsx
<InputGroup label="Username" error helperText="Username is already taken.">
  <Input placeholder="Choose a username" />
</InputGroup>
```
