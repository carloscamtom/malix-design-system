## Installation

```tsx
import { Checkbox } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
<Checkbox
  label="Accept terms and conditions"
  checked={false}
  onChange={(checked) => console.log(checked)}
/>
```

## Checked State

```tsx
<Checkbox
  label="Remember me"
  checked={true}
  onChange={(checked) => console.log(checked)}
/>
```

## Indeterminate State

Use the indeterminate state when a checkbox controls a group with mixed selections:

```tsx
<Checkbox
  label="Select all items"
  indeterminate
  onChange={(checked) => console.log(checked)}
/>
```

## Disabled State

```tsx
<Checkbox label="Read-only option" checked={true} disabled />
<Checkbox label="Unavailable option" checked={false} disabled />
```
