## Installation

```tsx
import { UserProfilePopover } from '@camtomlabs/malix-design-system';
```

## Usage

```tsx
import { Settings, CreditCard, HelpCircle } from 'lucide-react';

<UserProfilePopover
  name="Carlos Garcia"
  email="carlos@example.com"
  avatar={<Avatar initials="CG" size={40} />}
  credits={250}
  menuItems={[
    { icon: <Settings size={16} />, label: 'Settings', onClick: () => navigate('/settings') },
    { icon: <CreditCard size={16} />, label: 'Billing', onClick: () => navigate('/billing') },
    { icon: <HelpCircle size={16} />, label: 'Help', onClick: () => navigate('/help') },
  ]}
  onLogout={handleLogout}
/>
```

## Minimal

Without credits or custom menu items:

```tsx
<UserProfilePopover
  name="Jane Doe"
  email="jane@example.com"
  onLogout={handleLogout}
/>
```
